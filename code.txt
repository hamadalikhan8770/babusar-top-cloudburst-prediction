def focal_loss(gamma=2.0, alpha=0.5):
    """
    Focal loss for binary classification.
    gamma: focusing parameter; alpha: balancing factor.
    Adjusted alpha to 0.5 to give more weight to positive class.
    """
    def loss(y_true, y_pred):
        epsilon = K.epsilon()
        y_pred = K.clip(y_pred, epsilon, 1. - epsilon)
        pt = tf.where(tf.equal(y_true, 1), y_pred, 1 - y_pred)
        loss_val = -alpha * K.pow(1. - pt, gamma) * K.log(pt)
        return K.mean(loss_val)
    return loss

def create_sequences(group, seq_len, horizon, feature_cols, target_col):
    """
    For a single location group, create sequences of features (seq_len) and labels (horizon steps ahead).
    Returns X, y, and the valid_time of the target.
    """
    X, y, times = [], [], []
    features = group[feature_cols].values
    labels = group[target_col].values
    time_vals = group['valid_time'].values
    for i in range(len(group) - seq_len - horizon + 1):
        X.append(features[i:i+seq_len])
        y.append(labels[i+seq_len+horizon-1])
        times.append(time_vals[i+seq_len+horizon-1])
    return np.array(X), np.array(y), np.array(times)

def plot_training_history(history):
    """
    Plot training and validation loss and metrics.
    """
    fig, axes = plt.subplots(2, 2, figsize=(12,8))
    axes[0,0].plot(history.history['loss'], label='train')
    axes[0,0].plot(history.history['val_loss'], label='val')
    axes[0,0].set_title('Loss')
    axes[0,0].legend()

    axes[0,1].plot(history.history['auc'], label='train')
    axes[0,1].plot(history.history['val_auc'], label='val')
    axes[0,1].set_title('AUC')
    axes[0,1].legend()

    axes[1,0].plot(history.history['pr_auc'], label='train')
    axes[1,0].plot(history.history['val_pr_auc'], label='val')
    axes[1,0].set_title('PR AUC')
    axes[1,0].legend()

    axes[1,1].axis('off')
    plt.tight_layout()
    plt.show()

def evaluate_model(y_true, y_pred_prob, threshold=0.5):
    """
    Compute and print various metrics, return dictionary.
    """
    y_pred = (y_pred_prob > threshold).astype(int)
    accuracy = accuracy_score(y_true, y_pred)
    precision = precision_score(y_true, y_pred, zero_division=0)
    recall = recall_score(y_true, y_pred, zero_division=0)
    f1 = f1_score(y_true, y_pred, zero_division=0)
    roc_auc = roc_auc_score(y_true, y_pred_prob)
    precision_vals, recall_vals, _ = precision_recall_curve(y_true, y_pred_prob)
    pr_auc = auc(recall_vals, precision_vals)

    print(f"Accuracy:  {accuracy:.4f}")
    print(f"Precision: {precision:.4f}")
    print(f"Recall:    {recall:.4f}")
    print(f"F1 Score:  {f1:.4f}")
    print(f"ROC AUC:   {roc_auc:.4f}")
    print(f"PR AUC:    {pr_auc:.4f}")

    return {
        'accuracy': accuracy,
        'precision': precision,
        'recall': recall,
        'f1': f1,
        'roc_auc': roc_auc,
        'pr_auc': pr_auc
    }

def create_rolling_windows(years_array, y_array, min_pos=50, exclude_last_year=True):
    """
    Create rolling validation windows ensuring each validation set has at least min_pos positives.
    years_array: array of years for each sample.
    y_array: binary labels.
    exclude_last_year: if True, the last year in the dataset is reserved for final testing and not used in any validation window.
    Returns list of dicts with 'train_years' (list of years) and 'val_years' (list of years).
    """
    unique_years = np.sort(np.unique(years_array))
    if exclude_last_year:
        unique_years = unique_years[:-1]  # remove last year for test
    windows = []
    i = 0
    while i < len(unique_years):
        val_years = []
        cum_pos = 0
        j = i
        while j < len(unique_years) and cum_pos < min_pos:
            val_years.append(unique_years[j])
            mask = years_array == unique_years[j]
            cum_pos += y_array[mask].sum()
            j += 1
        if cum_pos >= min_pos:
            train_years = unique_years[:i].tolist()  # all years before the first validation year
            windows.append({
                'train_years': train_years,
                'val_years': val_years
            })
            i = j  # move to next year after this validation block
        else:
            # Not enough positives even after including all remaining years – this should be the last window
            train_years = unique_years[:i].tolist()
            val_years = unique_years[i:].tolist()
            windows.append({
                'train_years': train_years,
                'val_years': val_years
            })
            break
    return windows

def rolling_validation_score(hp, verbose=0):
    """
    Perform rolling window validation and return average validation PR-AUC.
    hp: hyperparameters dictionary.
    """
    scores = []
    for window in rolling_windows:
        train_mask = np.isin(years, window['train_years'])
        val_mask = np.isin(years, window['val_years'])
        if np.sum(val_mask) == 0:
            continue

        X_train_win = X[train_mask]
        y_train_win = y[train_mask]

        # Skip window if training data is empty
        if len(X_train_win) == 0:
            if verbose:
                print(f"Skipping window (train_years: {window['train_years']}, val_years: {window['val_years']}): Empty training set.")
            continue

        X_val_win = X[val_mask]
        y_val_win = y[val_mask]

        # Scale features
        n_samples_win, n_timesteps, n_features = X_train_win.shape
        X_train_2d = X_train_win.reshape(-1, n_features)
        X_val_2d = X_val_win.reshape(-1, n_features)

        scaler_win = StandardScaler()
        X_train_scaled_2d = scaler_win.fit_transform(X_train_2d)
        X_train_scaled = X_train_scaled_2d.reshape(n_samples_win, n_timesteps, n_features)

        X_val_scaled = scaler_win.transform(X_val_2d).reshape(X_val_win.shape[0], n_timesteps, n_features)

        # Compute class weights to handle imbalance (instead of SMOTE)
        # Use balanced class weights based on training labels
        class_weights = class_weight.compute_class_weight(
            class_weight='balanced',
            classes=np.unique(y_train_win),
            y=y_train_win
        )
        class_weight_dict = {i: class_weights[i] for i in range(len(class_weights))}

        # Build model with given hyperparameters
        model = build_model_from_hp(hp, n_timesteps, n_features)

        early_stop = EarlyStopping(monitor='val_pr_auc', mode='max', patience=7, restore_best_weights=True, verbose=0)  # increased patience
        reduce_lr = ReduceLROnPlateau(monitor='val_loss', factor=0.5, patience=3, min_lr=1e-6, verbose=0)

        history = model.fit(
            X_train_scaled, y_train_win,
            validation_data=(X_val_scaled, y_val_win),
            epochs=30,
            batch_size=64,
            class_weight=class_weight_dict,  # added class weights
            callbacks=[early_stop, reduce_lr],
            verbose=0
        )

        # Get best validation PR-AUC
        best_pr_auc = max(history.history['val_pr_auc'])
        scores.append(best_pr_auc)
        if verbose:
            print(f"Train years: {window['train_years']}, Val years: {window['val_years']}: PR-AUC = {best_pr_auc:.4f}")

    if len(scores) == 0:
        return 0.0
    return np.mean(scores)

def build_model_from_hp(hp, n_timesteps, n_features):
    """Build a simple Bidirectional LSTM model (no CNN, no attention)."""
    input_layer = Input(shape=(n_timesteps, n_features))

    # Single Bidirectional LSTM layer
    lstm_units = hp.get('lstm_units', 64)
    dropout_rate = hp.get('dropout', 0.3)
    x = Bidirectional(LSTM(lstm_units, return_sequences=False, dropout=dropout_rate))(input_layer)

    # Dense layers with L2 regularization
    dense_units = hp.get('dense_units', 32)
    x = Dense(dense_units, activation='relu', kernel_regularizer=l2(1e-4))(x)
    x = Dropout(dropout_rate)(x)

    output = Dense(1, activation='sigmoid')(x)

    model = Model(inputs=input_layer, outputs=output)
    lr = hp.get('lr', 0.001)
    model.compile(optimizer=Adam(learning_rate=lr),
                  loss=focal_loss(gamma=2.0, alpha=0.5),  # increased alpha to 0.5 for more focus on positives
                  metrics=[tf.keras.metrics.AUC(name='auc'),
                           tf.keras.metrics.AUC(name='pr_auc', curve='PR')])
    return model

# Random search with increased trials
import random

best_score = -1
best_hps = None
results = []

for trial in tqdm(range(n_trials)):
    # Sample hyperparameters
    hp = {}
    hp['conv_filters'] = random.choice(param_space['conv_filters'])    # dummy
    hp['kernel_size'] = random.choice(param_space['kernel_size'])      # dummy
    hp['second_conv'] = random.choice(param_space['second_conv'])      # dummy
    hp['conv_filters2'] = random.choice(param_space['conv_filters2'])  # dummy
    hp['lstm_units'] = random.choice(param_space['lstm_units'])
    hp['dropout'] = random.choice(param_space['dropout'])
    hp['dense_units'] = random.choice(param_space['dense_units'])
    hp['lr'] = random.choice(param_space['lr'])

    score = rolling_validation_score(hp, verbose=0)
    results.append((hp, score))
    if score > best_score:
        best_score = score
        best_hps = hp
    print(f"Trial {trial+1}: PR-AUC = {score:.4f}, best so far = {best_score:.4f}")
